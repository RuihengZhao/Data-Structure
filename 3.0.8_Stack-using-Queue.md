# Implement Stack using Queues

```
class Stack {
    Queue<Integer> q1 = new Queue<Integer>();
    Queue<Integer> q2 = new Queue<Integer>();

    int curr_size;

    Stack() {
        curr_size = 0;
    }

    void push(int x) {
        curr_size++;
        q1.add(x);
    }

    int pop() {
        int temp;
        while(true) {
            temp = q1.remove();
            if (q1.isEmpty()) break;
            q2.add(temp);
        }

        Queue<Integer> tempQ = q1;
        q1 = q2;
        q2 = tempQ;
        
        curr_size--;

        return temp;
    }
}
```
